275void MainWindow::on_StartBtn_clicked()
276{
277

304 this->ui->BlockSizeEdit->setTextInteractionFlags(Qt::NoTextInteraction);
305
307 bool directMode = this->ui->directModeCheckBox->isChecked();
309 bool report = this->ui->reportCheckBox->isChecked();
311 bool mt_enabled = this->ui->multithreadCheckBox->isChecked();
312
313 if(mt_enabled){
314
316 QThread *thread = QThread::create(&appManager::startBenchmarking, manager, directMode, report);
317
319 thread->start();
320
322 while(thread->isRunning()){
323
324#ifdef LINUX
325 usleep(1);
326#else
327 Sleep(1);
328#endif
329 qApp->processEvents();
330 updateProgressBar(manager->benchmarkProgress());
331 }
332 }
333
334 else{
335 manager->startBenchmarking(directMode); // single thread?
336 }
337 QMessageBox messageBox;
338 messageBox.information(0,"Successfully performed benchmarking","Benchmarking finished.");
339 }
340 catch(std::string &msg){
341 QMessageBox messageBox;
342 messageBox.critical(0,"Error","Unable to perform benchmark.\nOperation aborted.\nReason is \n{" + QString::fromStdString(msg) + "}");
343 restrictGUIElements(true);
344 }
345 catch(...)
346 {
347 QMessageBox messageBox;
348 messageBox.critical(0,"Error","Unable to perform benchmark.\nOperation aborted.");
349 restrictGUIElements(true);
